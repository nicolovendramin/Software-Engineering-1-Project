package it.polimi.ingsw.cg25.gui;

import java.awt.Color;
import java.awt.Font;
import java.awt.Image;
import java.io.File;
import java.io.IOException;

import javax.imageio.ImageIO;
import javax.swing.ImageIcon;
import javax.swing.JLabel;
import it.polimi.ingsw.cg25.model.dto.DTOPermitCard;

@SuppressWarnings("serial")
public class PermitCardLabel extends JLabel {

	public PermitCardLabel(DTOPermitCard permit, CoFGui gui) {
		try {
			ImageIcon icon = new ImageIcon(ImageIO.read(new File("src/main/resources/permitTop.png")).getScaledInstance(65, 65, Image.SCALE_SMOOTH));
			this.setIcon(icon);
		} catch (IOException e) {
			e.printStackTrace();
		}
		this.setBounds(0,0,65,95);
		this.setFont(gui.getFont());
		String c = permit.getCities().get(0).getName().substring(0, 1).toUpperCase();
		for(int i=1;i<permit.getCities().size();i++)
			c = c.concat(",").concat(permit.getCities().get(i).getName().substring(0, 1).toUpperCase());
		this.setText(c);
		this.setVerticalTextPosition(JLabel.TOP);
		this.setHorizontalTextPosition(JLabel.CENTER);
		this.setFont(new Font(gui.getFont().getFontName(), Font.PLAIN, 18));
		this.setForeground(Color.BLACK);
	}
}
