package it.polimi.ingsw.cg25.gui;

public class MapPoint {
	private final float x;
	private final float y;
	
	public MapPoint(float x,float y){
		this.x = x;
		this.y = y;
	}

	public float x(){
		return this.x;
	}
	
	public float y(){
		return this.y;
	}

	public String toString(){
		return " x : " + Float.toString(x) + " , y : " + Float.toString(y);
	}
}
