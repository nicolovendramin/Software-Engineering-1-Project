	public static void execution(int numberOfCitiesToPlaceInEachRegion,int numberOfRegions){
		Random random = new Random();
		for(int i = 0; i < numberOfRegions * numberOfCitiesToPlaceInEachRegion; i++){
			System.out.println("Locating point number " + Integer.toString(i));
			System.out.println(conditions);
			boolean ok = false;
			while(!ok)
			{
				random = new Random();
				float tempX = (random.nextFloat() % 1) * 1000;
				float tempY = (random.nextFloat() % 1 ) * 1000;
				tempX = tempX - (int)(tempX/80) * 80 + 10;
				tempY = tempY - (int)(tempY*numberOfRegions/100) *(100 / numberOfRegions) + (int)(100 / numberOfRegions)*(int)(i/(numberOfCitiesToPlaceInEachRegion*numberOfRegions));
				try{
					MapPoint p = new MapPoint(tempX,tempY);
					for(Line line : conditions){
						if(line.inLine(p))
							throw new IllegalArgumentException("I am " + p.toString() + " and I was in line with " + line.toString());
					}
					List<Line> tempConditions = new ArrayList<>();
					for(MapPoint point : cityLocations){
						tempConditions.add(new Line(p,point,4));
					}
					cityLocations.add(p);
					conditions.addAll(tempConditions);
					System.out.println("Located point # " + Integer.toString(i));
					JFrame f = new JFrame(Integer.toString(i));
					f.setSize(30, 30);
					f.setLocation((int)(p.x()/100 * Toolkit.getDefaultToolkit().getScreenSize().getHeight()),(int)(p.y()/100*(Toolkit.getDefaultToolkit().getScreenSize().getWidth())));
					f.setVisible(true);
					ok = true;
				}
				catch(IllegalArgumentException e){
					System.out.println(e.getMessage());
					ok = false;
				}
			}
		}
	}
	
	
	public static void main(String []args){
		Mapper.execution(5, 3);
