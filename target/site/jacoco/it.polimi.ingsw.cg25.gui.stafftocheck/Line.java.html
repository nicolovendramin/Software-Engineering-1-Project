package it.polimi.ingsw.cg25.gui.stafftocheck;

public class Line {
	private final float m;
	private final float q;
	private final float multiplier;
	
	public Line(MapPoint a,MapPoint b,float multiplier){
		if(samePoint(a,b)) throw new IllegalArgumentException("The two points are too close: " + a.toString() + " , " + b.toString());
		this.m = (b.y() - a.y()) / (b.x() - a.x());
		this.q = (b.x() * a.y() - a.x() * b.y()) / (b.x() - a.x());
		this.multiplier = multiplier;
	}
	
	public Line(MapPoint a,MapPoint b){
		this(a,b,1);
	}
	
	private boolean samePoint(MapPoint a,MapPoint b){
		return (Math.abs(a.x() - b.x())<5 && Math.abs(a.y() - b.y())<5);
	}
	
	public boolean inLine(MapPoint a){
		int value;
		if(m <= 1) 
			value = (int)(1*multiplier);
		else
			value = (int)(m*multiplier); 
		return Math.abs(a.x() * m + q - a.y()) < value;
	}
	
	public String toString(){
		return " y = " + Float.toString(m) + " * x + " + Float.toString(q);
	}
}
