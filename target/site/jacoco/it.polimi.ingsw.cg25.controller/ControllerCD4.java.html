package it.polimi.ingsw.cg25.controller;

import it.polimi.ingsw.cg25.actions.Action;
import it.polimi.ingsw.cg25.observer.Observable;
import it.polimi.ingsw.cg25.observer.Observer;
import it.polimi.ingsw.cg25.proxies.ViewProxy;

public class ControllerCD4 extends Observable<Action> implements Observer<Action> {

	private Action currentlyExecutingAction;
	private ViewProxy viewProxy;

	public ControllerCD4(Action startingAction) {
		this.currentlyExecutingAction = startingAction;
		this.viewProxy = new ViewProxy(this);
		//startingAction.getModel().attachObserver(this); TODO remove this comment
	}

	@Override
	public void update() {
		// this method is still empty because it is useless at the moment
	}

	@Override
	public void update(Action change) {
		Action c = change;
		this.currentlyExecutingAction = c;
		this.start();
	}

	public void sendReply(String s) {
		try {
			this.currentlyExecutingAction.sendReply(s);
			if (this.currentlyExecutingAction.hasNext())
				viewProxy.send(this.currentlyExecutingAction.next().printOptions());
			else {
				try {
					notifyObservers(this.currentlyExecutingAction);
				} catch (Exception e) {
					// here we have to catch the exception, advertise the error
					// to the view and reset the ChooseActionTypeAction from the
					// beginning
				}
			}
		} catch (Exception e) {
			this.viewProxy.send(e.getMessage());
			this.viewProxy.send(currentlyExecutingAction.next().printOptions());
		}
	}

	public Observer<String> getViewProxy() {
		return this.viewProxy;
	}

	public int attachView(Observer<String> e) {
		this.viewProxy.attachObserver(e);
		return ViewProxy.getID();
	}

	public void start() {
		if (currentlyExecutingAction.hasNext())
			viewProxy.send(this.currentlyExecutingAction.getInteraction(0).printOptions());
		else
			this.notifyObservers(currentlyExecutingAction);
	}

	public void removeView(Observer<String> e) {
		this.viewProxy.removeObserver(e);
	}
}
