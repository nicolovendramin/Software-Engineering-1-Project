package it.polimi.ingsw.cg25.actions;
import java.util.ArrayList;
import java.util.List;
/**
 * 
 * @author nicolo
 *
 */
public class ChooseActionTypeInteraction extends Interaction{

	private List<Action> choices;
	//index of the choices corresponding to the chosen one
	private Action reply;

	/**
	 * @param is the action to which the interaction is referred
	 * @param mainActionLeft number of main Action still to be done
	 * @param quickActionLeft number of quick Action that the player can still do
	 * @throws IllegalArgumentException when one of the parameters is negative
	 */
	public ChooseActionTypeInteraction(Action myAction,int mainActionLeft,int quickActionLeft)  
	{
		this.setMyAction(myAction);
		this.reply = null;
		this.choices = new ArrayList<>();
		super.setQuestion("Choose what you want to do:");
		if(myAction == null) 
			throw new IllegalArgumentException("The reference to the Match must not be null");
		if(mainActionLeft < 0 || quickActionLeft <0) 
			throw new IllegalArgumentException("Parameters can not be negatives");
		if(mainActionLeft != 0) 
			choices.add(new ChooseMainActionAction(myAction.getModel()));
		if(quickActionLeft != 0) 
			choices.add(new ChooseQuickActionAction(myAction.getModel()));
		if(myAction.getModel().canPass()) 
			choices.add(new PassAction(myAction.getModel()));
	}
	
	/**
	 * 
	 * @param question is the new question text
	 * @param mainActionLeft number of main Action still to be done
	 * @param quickActionLeft number of quick Action that the player can still do
	 * @throws IllegalArgumentException when one of the parameters is negative
	 */
	public ChooseActionTypeInteraction(Action myAction,String question,int mainActionLeft,int quickActionLeft)
	{
		this(myAction,mainActionLeft,quickActionLeft);
		super.setQuestion(question);
	}
	
	/**
	 * This method is called to parse the reply to the interaction from a string
	 * In the string is expected to be encoded the index of the List choices corresponding
	 * to the choice of the user
	 * @throws IllegalARgumentException when either the input is not int-parsable or it does not make sense
	 */
	@Override
	public void registerReply(String reply) {
		int index = 0;
		try
		{
			index = Integer.parseInt(reply);
			this.reply = choices.get(index);
		}
		catch(Exception e)
		{
			throw new IllegalArgumentException("The given input is not valid");
		}
			
		
	}

	/**
	 * Prints a String representation of the interaction including both the question and the answer
	 */
	@Override
	public String printOptions() {
		StringBuilder token = new StringBuilder(super.getQuestion());
		for(int i=0;i<choices.size();i++)
		{
			token.append("\n").append(Integer.toString(i)).append(")").append(choices.get(i).toString());
		}
		return token.toString();
	}

	/**
	 * Simple getter for the Reply field
	 * @return the reply registered for the current Interaction
	 */
	public Action getReply() 
	{
		if(reply == null) 
			throw new NullPointerException("You must have a reply registered before try to access it");
		return this.reply;
	}
	

}
