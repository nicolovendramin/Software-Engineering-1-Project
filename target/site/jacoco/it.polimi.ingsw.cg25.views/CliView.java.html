package it.polimi.ingsw.cg25.views;
import it.polimi.ingsw.cg25.actions.Interaction;
import it.polimi.ingsw.cg25.communication.Packet;
import it.polimi.ingsw.cg25.communication.VectorPacket;
import it.polimi.ingsw.cg25.observer.Observable;
import it.polimi.ingsw.cg25.observer.Observer;

public class CliView extends Observable<VectorPacket<Interaction>> implements View, Observer<Packet<Interaction>>{

	private final int myId;
	private Interaction myActionInteraction;
	
	public CliView(int id){
		this.myId = id;
	}
	
	public void input(String s) {
		String[] splitted = s.split(":")[1].split("]");
		if(splitted[0].equals("[Action"))
		{
			if(myActionInteraction == null) 
				System.out.println("not your turn baby");
			else 
				try{
					myActionInteraction.registerReply(splitted[1]);
					this.notifyObservers(new VectorPacket<Interaction>(myActionInteraction,"Action",0,this.myId));
				}catch(Exception e){
					System.out.println(e.getMessage());
				}
		}
	}

	@Override
	public void update() {
		// TODO Auto-generated method stub
		
	}

	@Override
	public void update(Packet<Interaction> change) {
		try{
			Interaction toBeChecked = change.getContent(this.myId);
			myActionInteraction = toBeChecked;
			System.out.println(Integer.toString(myId) + "[".concat(change.getType()).concat("]").concat(myActionInteraction.printOptions()));
		}catch(IllegalAccessException e){
			//means that I was not meant to read the message
		}
	}

}
