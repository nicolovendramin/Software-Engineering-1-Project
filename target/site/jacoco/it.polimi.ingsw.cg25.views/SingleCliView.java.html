package it.polimi.ingsw.cg25.views;

import it.polimi.ingsw.cg25.controller.ActionController;
import it.polimi.ingsw.cg25.observer.Observable;
import it.polimi.ingsw.cg25.observer.Observer;

public class SingleCliView extends Observable<String> implements View, Observer<String>{

	private ActionController controller;
	private int id;
	
	public SingleCliView(ActionController controller)
	{
		this.controller = controller;
		this.id = this.controller.attachView(this);
	}
	
	@Override
	public void update() {
	}

	@Override
	public void update(String change) {
		if(!isForMe(change,this.id)) return;
		System.out.println(change);
		if(isQuestion(change)) 
		{
			CliInputThread getReply = new CliInputThread(this);
			getReply.start();
		}
	}
	

	
	public void input(String s)
	{
		notifyObservers(Integer.toString(this.id).concat("||").concat(s));
	}
}
