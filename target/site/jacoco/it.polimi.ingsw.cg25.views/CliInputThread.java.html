package it.polimi.ingsw.cg25.views;
import java.util.List;
import java.util.Scanner;

public class CliInputThread extends InputThread{

	List<CliView> view;
	
	public CliInputThread(List<CliView> view){
		this.view = view;
		System.out.println(this.view.size());
	}
	
	@Override
	public void run() {
		Scanner scan = new Scanner(System.in);
		while(view!=null){
		if(scan.hasNextLine())
		{	
			String s = scan.nextLine();
			this.view.get(Integer.parseInt(s.split(":")[0])).input(s);
			System.out.println(s);
		}
		}
		scan.close();
	}
	
}
