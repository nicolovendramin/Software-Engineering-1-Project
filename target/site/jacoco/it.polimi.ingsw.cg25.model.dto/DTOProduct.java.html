package it.polimi.ingsw.cg25.model.dto;

import java.util.ArrayList;
import java.util.List;

import it.polimi.ingsw.cg25.exceptions.ElementNotFoundException;
import it.polimi.ingsw.cg25.model.trade.Product;
import it.polimi.ingsw.cg25.model.trade.ProductBonus;
import it.polimi.ingsw.cg25.model.trade.ProductPermitCard;
import it.polimi.ingsw.cg25.model.trade.ProductPoliticsCard;
import it.polimi.ingsw.cg25.model.trade.Sellable;

public abstract class DTOProduct<T extends DTOSellable> implements DTO {

	/**
	 * Serial number for serial communication
	 */
	private static final long serialVersionUID = 8168423406926931295L;
	/**
	 * The price of the DTO product
	 */
	final private int price;
	/**
	 * The DTO player who is the owner of the product and wants to sell it
	 */
	final private DTOPlayerCD4 owner;
	
	final private List<T> itemsToSell;
	
	public DTOProduct(int price, DTOPlayerCD4 owner, List<T> itemsToSell) {
		this.price = price;
		this.owner = owner;
		this.itemsToSell = itemsToSell;
	}
	
	private static DTOProduct<? extends DTOSellable> map(Product<? extends Sellable> product) {
		if(product instanceof ProductBonus)
			return new DTOProductBonus((ProductBonus) product);
		if(product instanceof ProductPermitCard)
			return new DTOProductPermit((ProductPermitCard) product);
		if(product instanceof ProductPoliticsCard)
			return new DTOProductPoliticsCard((ProductPoliticsCard) product);
		throw new IllegalArgumentException();
	}
	
	public static DTOProduct<? extends DTOSellable> convert(Product<? extends Sellable> toConvert) {
		DTOProduct<? extends DTOSellable> converted = DTOProduct.map(toConvert);
		return converted;
	}
	
	public static List<DTOProduct<? extends DTOSellable>> convertAll(List<Product<? extends Sellable>> toConvert) {
		List<DTOProduct<? extends DTOSellable>> converted = new ArrayList<>();
		toConvert.forEach(p -> converted.add(DTOProduct.convert(p)));
		
		return converted;
	}

	@Override
	public int hashCode() {
		final int prime = 31;
		int result = 1;
		result = prime * result + ((itemsToSell == null) ? 0 : itemsToSell.hashCode());
		result = prime * result + ((owner == null) ? 0 : owner.hashCode());
		result = prime * result + price;
		return result;
	}

	@Override
	public boolean equals(Object obj) {
		if(obj == null)
			return false;
		//Se obj è un DTOProduct...
		if(obj instanceof DTOProductBonus) {
			DTOProductBonus converted = (DTOProductBonus)obj;
			return this.getClass() == DTOProductBonus.class && this.price == converted.getPrice() &&
					this.owner.equals(converted.getOwner()) && this.itemsToSell.equals(converted.getItemsToSell());
		}
		else if(obj instanceof DTOProductPermit) {
			DTOProductPermit converted = (DTOProductPermit)obj;
			return this.getClass() == DTOProductPermit.class && this.price == converted.getPrice() &&
					this.owner.equals(converted.getOwner()) && this.itemsToSell.equals(converted.getItemsToSell());
		}
		else if(obj instanceof DTOProductPoliticsCard) {
			DTOProductPoliticsCard converted = (DTOProductPoliticsCard)obj;
			return this.getClass() == DTOProductPoliticsCard.class && this.price == converted.getPrice() &&
					this.owner.equals(converted.getOwner()) && this.itemsToSell.equals(converted.getItemsToSell());
		}
		//Se obj è un Product...
		else if(obj instanceof ProductBonus) {
			ProductBonus converted = (ProductBonus)obj;
			return this.price == converted.getPrice() && this.owner.equals(converted.getOwner()) &&
					this.itemsToSell.equals(converted.getContent());
		}
		else if(obj instanceof ProductPermitCard) {
			ProductPermitCard converted = (ProductPermitCard)obj;
			return this.price == converted.getPrice() && this.owner.equals(converted.getOwner()) &&
					this.itemsToSell.equals(converted.getContent());
		}
		else if(obj instanceof ProductPoliticsCard) {
			ProductPoliticsCard converted = (ProductPoliticsCard)obj;
			return this.price == converted.getPrice() && this.owner.equals(converted.getOwner()) &&
					this.itemsToSell.equals(converted.getContent());
		}
		return false;
	}
	
	public Product<? extends Sellable> decode(List<Product<? extends Sellable>> candidates) throws ElementNotFoundException {
		for(Product<? extends Sellable> p : candidates)
			if(this.equals(p))
				return p;
		throw new ElementNotFoundException();
	}
	
	public static List<Product<? extends Sellable>> decodeAll(List<DTOProduct<? extends DTOSellable>> toDecode, 
			List<Product<? extends Sellable>> candidates) throws ElementNotFoundException {
		
		List<DTOProduct<? extends DTOSellable>> toDecodeCopy = new ArrayList<>(toDecode);
		List<Product<? extends Sellable>> decoded = new ArrayList<>();
		List<Product<? extends Sellable>> candidateCopy = new ArrayList<>(candidates);
		for(DTOProduct<? extends DTOSellable> dc : toDecodeCopy){
			Product<? extends Sellable> temp = dc.decode(candidateCopy);
			decoded.add(temp);
			candidateCopy.remove(temp);
			}
		return decoded;
	}

	public int getPrice() {
		return price;
	}

	public DTOPlayerCD4 getOwner() {
		return owner;
	}

	public List<T> getItemsToSell() {
		return itemsToSell;
	}
	
}
